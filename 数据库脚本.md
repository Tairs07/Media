1. 创建 Users 表
```sql
  -- ============================================
  -- 创建 Users 表
  -- ============================================
  CREATE TABLE "Users" (
      "Id" SERIAL PRIMARY KEY,
      "Username" VARCHAR(50) NOT NULL,
      "Email" VARCHAR(100) NOT NULL,
      "PasswordHash" VARCHAR(255) NOT NULL,
      "AvatarUrl" VARCHAR(500),
      "Bio" VARCHAR(500),
      "CreatedAt" TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
      "UpdatedAt" TIMESTAMP,
      "IsActive" BOOLEAN NOT NULL DEFAULT true
  );

  -- 创建唯一索引
  CREATE UNIQUE INDEX "IX_Users_Username" ON "Users" ("Username");
  CREATE UNIQUE INDEX "IX_Users_Email" ON "Users" ("Email");

  -- 添加注释
  COMMENT ON TABLE "Users" IS '用户表';
  COMMENT ON COLUMN "Users"."Id" IS '用户ID';
  COMMENT ON COLUMN "Users"."Username" IS '用户名';
  COMMENT ON COLUMN "Users"."Email" IS '邮箱';
  COMMENT ON COLUMN "Users"."PasswordHash" IS '密码哈希';
  COMMENT ON COLUMN "Users"."AvatarUrl" IS '头像URL';
  COMMENT ON COLUMN "Users"."Bio" IS '个人简介';
  COMMENT ON COLUMN "Users"."CreatedAt" IS '创建时间';
  COMMENT ON COLUMN "Users"."UpdatedAt" IS '更新时间';
  COMMENT ON COLUMN "Users"."IsActive" IS '是否激活';
```


  2. 创建 MediaFiles 表
```sql
  -- ============================================
  -- 创建 MediaFiles 表
  -- ============================================
  CREATE TABLE "MediaFiles" (
      "Id" SERIAL PRIMARY KEY,
      "Title" VARCHAR(200) NOT NULL,
      "FileName" VARCHAR(255) NOT NULL,
      "FileType" VARCHAR(50) NOT NULL,
      "MimeType" VARCHAR(100) NOT NULL,
      "FilePath" VARCHAR(500) NOT NULL,
      "ThumbnailPath" VARCHAR(500),
      "FileSize" BIGINT NOT NULL,
      "Width" INTEGER,
      "Height" INTEGER,
      "Duration" INTEGER,
      "UploadedAt" TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
      "UpdatedAt" TIMESTAMP,
      "Description" VARCHAR(500),
      "Tags" VARCHAR(100),
      "ViewCount" INTEGER NOT NULL DEFAULT 0,
      "LikeCount" INTEGER NOT NULL DEFAULT 0,
      "IsPublic" BOOLEAN NOT NULL DEFAULT true,
      "UserId" INTEGER NOT NULL,

      -- 外键约束
      CONSTRAINT "FK_MediaFiles_Users" FOREIGN KEY ("UserId")
          REFERENCES "Users" ("Id")
          ON DELETE CASCADE
  );

  -- 创建索引
  CREATE INDEX "IX_MediaFiles_UserId" ON "MediaFiles" ("UserId");
  CREATE INDEX "IX_MediaFiles_FileType" ON "MediaFiles" ("FileType");
  CREATE INDEX "IX_MediaFiles_UploadedAt" ON "MediaFiles" ("UploadedAt");
  CREATE INDEX "IX_MediaFiles_UserId_IsPublic" ON "MediaFiles" ("UserId", "IsPublic");

  -- 添加注释
  COMMENT ON TABLE "MediaFiles" IS '媒体文件表';
  COMMENT ON COLUMN "MediaFiles"."Id" IS '媒体ID';
  COMMENT ON COLUMN "MediaFiles"."Title" IS '媒体标题';
  COMMENT ON COLUMN "MediaFiles"."FileName" IS '文件名';
  COMMENT ON COLUMN "MediaFiles"."FileType" IS '文件类型(image/video)';
  COMMENT ON COLUMN "MediaFiles"."MimeType" IS 'MIME类型';
  COMMENT ON COLUMN "MediaFiles"."FilePath" IS '文件路径';
  COMMENT ON COLUMN "MediaFiles"."ThumbnailPath" IS '缩略图路径';
  COMMENT ON COLUMN "MediaFiles"."FileSize" IS '文件大小(字节)';
  COMMENT ON COLUMN "MediaFiles"."Width" IS '宽度(像素)';
  COMMENT ON COLUMN "MediaFiles"."Height" IS '高度(像素)';
  COMMENT ON COLUMN "MediaFiles"."Duration" IS '视频时长(秒)';
  COMMENT ON COLUMN "MediaFiles"."UploadedAt" IS '上传时间';
  COMMENT ON COLUMN "MediaFiles"."UpdatedAt" IS '更新时间';
  COMMENT ON COLUMN "MediaFiles"."Description" IS '描述';
  COMMENT ON COLUMN "MediaFiles"."Tags" IS '标签(逗号分隔)';
  COMMENT ON COLUMN "MediaFiles"."ViewCount" IS '浏览次数';
  COMMENT ON COLUMN "MediaFiles"."LikeCount" IS '点赞数';
  COMMENT ON COLUMN "MediaFiles"."IsPublic" IS '是否公开';
  COMMENT ON COLUMN "MediaFiles"."UserId" IS '用户ID';
```
  3. 创建 UserSessions 表
```sql
  -- ============================================
  -- 创建 UserSessions 表
  -- ============================================
  CREATE TABLE "UserSessions" (
      "Id" SERIAL PRIMARY KEY,
      "UserId" INTEGER NOT NULL,
      "Token" VARCHAR(500) NOT NULL,
      "IpAddress" VARCHAR(45) NOT NULL,
      "UserAgent" VARCHAR(500),
      "CreatedAt" TIMESTAMP NOT NULL DEFAULT CURRENT_TIMESTAMP,
      "ExpiresAt" TIMESTAMP NOT NULL,
      "IsActive" BOOLEAN NOT NULL DEFAULT true,

      -- 外键约束
      CONSTRAINT "FK_UserSessions_Users" FOREIGN KEY ("UserId")
          REFERENCES "Users" ("Id")
          ON DELETE CASCADE
  );

  -- 创建索引
  CREATE INDEX "IX_UserSessions_UserId" ON "UserSessions" ("UserId");
  CREATE UNIQUE INDEX "IX_UserSessions_Token" ON "UserSessions" ("Token");
  CREATE INDEX "IX_UserSessions_UserId_IsActive" ON "UserSessions" ("UserId", "IsActive");

  -- 添加注释
  COMMENT ON TABLE "UserSessions" IS '用户会话表';
  COMMENT ON COLUMN "UserSessions"."Id" IS '会话ID';
  COMMENT ON COLUMN "UserSessions"."UserId" IS '用户ID';
  COMMENT ON COLUMN "UserSessions"."Token" IS 'JWT令牌';
  COMMENT ON COLUMN "UserSessions"."IpAddress" IS 'IP地址';
  COMMENT ON COLUMN "UserSessions"."UserAgent" IS '浏览器标识';
  COMMENT ON COLUMN "UserSessions"."CreatedAt" IS '创建时间';
  COMMENT ON COLUMN "UserSessions"."ExpiresAt" IS '过期时间';
  COMMENT ON COLUMN "UserSessions"."IsActive" IS '是否活跃';
```
  执行顺序

  在 Navicat 中按以下顺序执行：

  1. 先执行脚本1 - 创建 Users 表（基础表，被其他表引用）
  2. 再执行脚本2 - 创建 MediaFiles 表（依赖 Users）
  3. 最后执行脚本3 - 创建 UserSessions 表（依赖 Users）

  验证表是否创建成功

  执行完三个脚本后，运行以下 SQL 验证：
```sql
  -- 查看所有表
  SELECT table_name
  FROM information_schema.tables
  WHERE table_schema = 'public';

  -- 查看表结构
  SELECT column_name, data_type, character_maximum_length, is_nullable, column_default
  FROM information_schema.columns
  WHERE table_name = 'Users'
  ORDER BY ordinal_position;
  ```
